countdownCube

Example 1

      $('#counter-days-only').countdownCube( {
        target: new Date( 'March 1, 2018 10:00:00' ),
        cubeSize: 150,
        background:  '#ffffff',
        color: 'red',
        labelsTranslations: {'year': 'anni',
                             'month': 'mesi',
                             'day': 'giorni',
                             'hour': 'ore',
                             'minute': 'minuti',
                             'second': 'secondi'
                             },
        showDaysOnly: true,
        } );
